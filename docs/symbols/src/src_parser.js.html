 27   function normalizeAttr(attr) {
 28     // transform attribute names

 35   /**
 36    * Returns an object of attributes' name/value, given element and an array of attribute names;
 37    * Parses parent "g" nodes recursively upwards.
 38    * @static
 39    * @memberOf fabric
 40    * @method parseAttributes
 41    * @param {DOMElement} element Element to parse
 42    * @param {Array} attributes Array of attributes to parse
 43    * @return {Object} object containing parsed attributes' names/values
 44    */
 45   function parseAttributes(element, attributes) {
 46 
 47     if (!element) {
 48       return;
 49     }
 50 
 51     var value,
 52         parsed,
 53         parentAttributes = { };
 54 
 55     // if there's a parent container (`g` node), parse its attributes recursively upwards
 56     if (element.parentNode && /^g$/i.test(element.parentNode.nodeName)) {
 57       parentAttributes = fabric.parseAttributes(element.parentNode, attributes);
 58     }
 59 
 60     var ownAttributes = attributes.reduce(function(memo, attr) {
 61       value = element.getAttribute(attr);
 62       parsed = parseFloat(value);
 63       if (value) {
 64         // "normalize" attribute values
 65         if ((attr === 'fill' || attr === 'stroke') && value === 'none') {
 66           value = '';
 67         }
 68         if (attr === 'fill-rule') {
 69           value = (value === 'evenodd') ? 'destination-over' : value;
 70         }
 71         if (attr === 'transform') {
 72           value = fabric.parseTransformAttribute(value);
 73         }
 74         attr = normalizeAttr(attr);
 75         memo[attr] = isNaN(parsed) ? value : parsed;
 76       }
 77       return memo;
 78     }, { });
 79 
 80     // add values parsed from style, which take precedence over attributes
 81     // (see: http://www.w3.org/TR/SVG/styling.html#UsingPresentationAttributes)
 82 
 83     ownAttributes = extend(ownAttributes, extend(getGlobalStylesForElement(element), fabric.parseStyleAttribute(element)));
 84     return extend(parentAttributes, ownAttributes);
 85   };

 87   /**
 88    * Parses "transform" attribute, returning an array of values
 89    * @static
 90    * @function
 91    * @memberOf fabric
 92    * @method parseTransformAttribute
 93    * @param attributeValue {String} string containing attribute value
 94    * @return {Array} array of 6 elements representing transformation matrix
 95    */
 96   fabric.parseTransformAttribute = (function() {
 97     function rotateMatrix(matrix, args) {
 98       var angle = args[0];
 99 
100       matrix[0] = Math.cos(angle);
101       matrix[1] = Math.sin(angle);
102       matrix[2] = -Math.sin(angle);
103       matrix[3] = Math.cos(angle);
104     }
105 
106     function scaleMatrix(matrix, args) {
107       var multiplierX = args[0],
108           multiplierY = (args.length === 2) ? args[1] : args[0];
109 
110       matrix[0] = multiplierX;
111       matrix[3] = multiplierY;
112     }
113 
114     function skewXMatrix(matrix, args) {
115       matrix[2] = args[0];
116     }
117 
118     function skewYMatrix(matrix, args) {
119       matrix[1] = args[0];
120     }
121 
122     function translateMatrix(matrix, args) {
123       matrix[4] = args[0];
124       if (args.length === 2) {
125         matrix[5] = args[1];
126       }
127     }
128 
129     // identity matrix
130     var iMatrix = [
131           1, // a
132           0, // b
133           0, // c
134           1, // d
135           0, // e
136           0  // f
137         ],
138 
139         // == begin transform regexp
140         number = '(?:[-+]?\\d+(?:\\.\\d+)?(?:e[-+]?\\d+)?)',
141         comma_wsp = '(?:\\s+,?\\s*|,\\s*)',
142 
143         skewX = '(?:(skewX)\\s*\\(\\s*(' + number + ')\\s*\\))',
144         skewY = '(?:(skewY)\\s*\\(\\s*(' + number + ')\\s*\\))',
145         rotate = '(?:(rotate)\\s*\\(\\s*(' + number + ')(?:' + comma_wsp + '(' + number + ')' + comma_wsp + '(' + number + '))?\\s*\\))',
146         scale = '(?:(scale)\\s*\\(\\s*(' + number + ')(?:' + comma_wsp + '(' + number + '))?\\s*\\))',
147         translate = '(?:(translate)\\s*\\(\\s*(' + number + ')(?:' + comma_wsp + '(' + number + '))?\\s*\\))',
148 
149         matrix = '(?:(matrix)\\s*\\(\\s*' +
150                   '(' + number + ')' + comma_wsp +
151                   '(' + number + ')' + comma_wsp +
152                   '(' + number + ')' + comma_wsp +
153                   '(' + number + ')' + comma_wsp +
154                   '(' + number + ')' + comma_wsp +
155                   '(' + number + ')' +
156                   '\\s*\\))',
157 
158         transform = '(?:' +
159                     matrix + '|' +
160                     translate + '|' +
161                     scale + '|' +
162                     rotate + '|' +
163                     skewX + '|' +
164                     skewY +
165                     ')',
166 
167         transforms = '(?:' + transform + '(?:' + comma_wsp + transform + ')*' + ')',
168 
169         transform_list = '^\\s*(?:' + transforms + '?)\\s*$',
170 
171         // http://www.w3.org/TR/SVG/coords.html#TransformAttribute
172         reTransformList = new RegExp(transform_list),
173         // == end transform regexp
174 
175         reTransform = new RegExp(transform);
176 
177     return function(attributeValue) {
178 
179       // start with identity matrix
180       var matrix = iMatrix.concat();
181 
182       // return if no argument was given or
183       // an argument does not match transform attribute regexp
184       if (!attributeValue || (attributeValue && !reTransformList.test(attributeValue))) {
185         return matrix;
186       }
187 
188       attributeValue.replace(reTransform, function(match) {
189 
190         var m = new RegExp(transform).exec(match).filter(function (match) {
191               return (match !== '' && match != null);
192             }),
193             operation = m[1],
194             args = m.slice(2).map(parseFloat);
195 
196         switch(operation) {
197           case 'translate':
198             translateMatrix(matrix, args);
199             break;
200           case 'rotate':
201             rotateMatrix(matrix, args);
202             break;
203           case 'scale':
204             scaleMatrix(matrix, args);
205             break;
206           case 'skewX':
207             skewXMatrix(matrix, args);
208             break;
209           case 'skewY':
210             skewYMatrix(matrix, args);
211             break;
212           case 'matrix':
213             matrix = args;
214             break;
215         }
216       })
217       return matrix;
218     }
219   })();

221   /**
222    * Parses "points" attribute, returning an array of values
223    * @static
224    * @memberOf fabric
225    * @method parsePointsAttribute
226    * @param points {String} points attribute string
227    * @return {Array} array of points
228    */
229   function parsePointsAttribute(points) {
230 
231     // points attribute is required and must not be empty
232     if (!points) return null;
233 
234     points = points.trim();
235     var asPairs = points.indexOf(',') > -1;
236 
237     points = points.split(/\s+/);
238     var parsedPoints = [ ];
239 
240     // points could look like "10,20 30,40" or "10 20 30 40"
241     if (asPairs) {
242      for (var i = 0, len = points.length; i < len; i++) {
243        var pair = points[i].split(',');
244        parsedPoints.push({ x: parseFloat(pair[0]), y: parseFloat(pair[1]) });
245      }
246     }
247     else {
248       for (var i = 0, len = points.length; i < len; i+=2) {
249         parsedPoints.push({ x: parseFloat(points[i]), y: parseFloat(points[i+1]) });
250       }
251     }
252 
253     // odd number of points is an error
254     if (parsedPoints.length % 2 !== 0) {
255       // return null;
256     }
257 
258     return parsedPoints;
259   };

261   /**
262    * Parses "style" attribute, retuning an object with values
263    * @static
264    * @memberOf fabric
265    * @method parseStyleAttribute
266    * @param {SVGElement} element Element to parse
267    * @return {Object} Objects with values parsed from style attribute of an element
268    */
269   function parseStyleAttribute(element) {
270     var oStyle = { },
271         style = element.getAttribute('style');
272     if (style) {
273       if (typeof style == 'string') {
274         style = style.replace(/;$/, '').split(';').forEach(function (current) {
275             var attr = current.split(':');
276             oStyle[normalizeAttr(attr[0].trim().toLowerCase())] = attr[1].trim();
277         });
278       } else {
279         for (var prop in style) {
280           if (typeof style[prop] !== 'undefined') {
281             oStyle[normalizeAttr(prop.toLowerCase())] = style[prop];
282           }
283         }
284       }
285     }
286     return oStyle;
287   };

289   function resolveGradients(instances) {
290     var activeInstance = fabric.Canvas.activeInstance,
291         ctx = activeInstance ? activeInstance.getContext() : null;
292 
293     if (!ctx) return;
294 
295     for (var i = instances.length; i--; ) {
296       var instanceFillValue = instances[i].get('fill');
297 
298       if (/^url\(/.test(instanceFillValue)) {
299 
300         var gradientId = instanceFillValue.slice(5, instanceFillValue.length - 1);
301 
302         if (fabric.gradientDefs[gradientId]) {
303           instances[i].set('fill',
304             fabric.Gradient.fromElement(fabric.gradientDefs[gradientId], ctx, instances[i]));
305         }
306       }
307     }

310   /**
311    * Transforms an array of svg elements to corresponding fabric.* instances
312    * @static
313    * @memberOf fabric
314    * @method parseElements
315    * @param {Array} elements Array of elements to parse
316    * @param {Function} callback Being passed an array of fabric instances (transformed from SVG elements)
317    * @param {Object} options Options object
318    */
319   function parseElements(elements, callback, options) {
320     var instances = Array(elements.length), i = elements.length;
321 
322     function checkIfDone() {
323       if (--i === 0) {
324         instances = instances.filter(function(el) {
325           return el != null;
326         });
327         resolveGradients(instances);
328         callback(instances);
329       }
330     }
331 
332     for (var index = 0, el, len = elements.length; index < len; index++) {
333       el = elements[index];
334       var klass = fabric[capitalize(el.tagName)];
335       if (klass && klass.fromElement) {
336         try {
337           if (klass.async) {
338             klass.fromElement(el, (function(index) {
339               return function(obj) {
340                 instances.splice(index, 0, obj);
341                 checkIfDone();
342               };
343             })(index), options);
344           }
345           else {
346             instances.splice(index, 0, klass.fromElement(el, options));
347             checkIfDone();
348           }
349         }
350         catch(e) {
351           fabric.log(e.message || e);
352         }
353       }
354       else {
355         checkIfDone();
356       }
357     }
358   };

360   /**
361    * Returns CSS rules for a given SVG document
362    * @static
363    * @function
364    * @memberOf fabric
365    * @method getCSSRules
366    * @param {SVGDocument} doc SVG document to parse
367    * @return {Object} CSS rules of this document
368    */
369   function getCSSRules(doc) {
370     var styles = doc.getElementsByTagName('style'),
371         allRules = { },
372         rules;
373 
374     // very crude parsing of style contents
375     for (var i = 0, len = styles.length; i < len; i++) {
376       var styleContents = styles[0].textContent;
377 
378       // remove comments
379       styleContents = styleContents.replace(/\/\*[\s\S]*?\*\//g, '');
380 
381       rules = styleContents.match(/[^{]*\{[\s\S]*?\}/g);
382       rules = rules.map(function(rule) { return rule.trim() });
383 
384       rules.forEach(function(rule) {
385         var match = rule.match(/([\s\S]*?)\s*\{([^}]*)\}/),
386             rule = match[1],
387             declaration = match[2].trim(),
388             propertyValuePairs = declaration.replace(/;$/, '').split(/\s*;\s*/);
389 
390         if (!allRules[rule]) {
391           allRules[rule] = { };
392         }
393 
394         for (var i = 0, len = propertyValuePairs.length; i < len; i++) {
395           var pair = propertyValuePairs[i].split(/\s*:\s*/),
396               property = pair[0],
397               value = pair[1];
398 
399           allRules[rule][property] = value;
400         }
401       });
402     }
403 
404     return allRules;
405   }

407   function getGlobalStylesForElement(element) {
408     var nodeName = element.nodeName,
409         className = element.getAttribute('class'),
410         id = element.getAttribute('id'),
411         styles = { };
412 
413     for (var rule in fabric.cssRules) {
414       var ruleMatchesElement = (className && new RegExp('^\\.' + className).test(rule)) ||
415                                (id && new RegExp('^#' + id).test(rule)) ||
416                                (new RegExp('^' + nodeName).test(rule));
417 
418       if (ruleMatchesElement) {
419         for (var property in fabric.cssRules[rule]) {
420           styles[property] = fabric.cssRules[rule][property];
421         }
422       }
423     }
424 
425     return styles;
426   }

428   /**
429    * Parses an SVG document, converts it to an array of corresponding fabric.* instances and passes them to a callback
430    * @static
431    * @function
432    * @memberOf fabric
433    * @method parseSVGDocument
434    * @param {SVGDocument} doc SVG document to parse
435    * @param {Function} callback Callback to call when parsing is finished; It's being passed an array of elements (parsed from a document).
436    */
437   fabric.parseSVGDocument = (function() {
438 
439     var reAllowedSVGTagNames = /^(path|circle|polygon|polyline|ellipse|rect|line|image)$/;
440 
441     // http://www.w3.org/TR/SVG/coords.html#ViewBoxAttribute
442     // \d doesn't quite cut it (as we need to match an actual float number)
443 
444     // matches, e.g.: +14.56e-12, etc.
445     var reNum = '(?:[-+]?\\d+(?:\\.\\d+)?(?:e[-+]?\\d+)?)';
446 
447     var reViewBoxAttrValue = new RegExp(
448       '^' +
449       '\\s*(' + reNum + '+)\\s*,?' +
450       '\\s*(' + reNum + '+)\\s*,?' +
451       '\\s*(' + reNum + '+)\\s*,?' +
452       '\\s*(' + reNum + '+)\\s*' +
453       '$'
454     );
455 
456     function hasAncestorWithNodeName(element, nodeName) {
457       while (element && (element = element.parentNode)) {
458         if (nodeName.test(element.nodeName)) {
459           return true;
460         }
461       }
462       return false;
463     }
464 
465     return function(doc, callback) {
466       if (!doc) return;
467 
468       var startTime = new Date(),
469           descendants = fabric.util.toArray(doc.getElementsByTagName('*'));
470 
471       if (descendants.length === 0) {
472         // we're likely in node, where "o3-xml" library fails to gEBTN("*")
473         // https://github.com/ajaxorg/node-o3-xml/issues/21
474         descendants = doc.selectNodes("//*[name(.)!='svg']");
475         var arr = [ ];
476         for (var i = 0, len = descendants.length; i < len; i++) {
477           arr[i] = descendants[i];
478         }
479         descendants = arr;
480       }
481 
482       var elements = descendants.filter(function(el) {
483         return reAllowedSVGTagNames.test(el.tagName) &&
484               !hasAncestorWithNodeName(el, /^(?:pattern|defs)$/); // http://www.w3.org/TR/SVG/struct.html#DefsElement
485       });
486 
487       if (!elements || (elements && !elements.length)) return;
488 
489       var viewBoxAttr = doc.getAttribute('viewBox'),
490           widthAttr = doc.getAttribute('width'),
491           heightAttr = doc.getAttribute('height'),
492           width = null,
493           height = null,
494           minX,
495           minY;
496 
497       if (viewBoxAttr && (viewBoxAttr = viewBoxAttr.match(reViewBoxAttrValue))) {
498         minX = parseInt(viewBoxAttr[1], 10);
499         minY = parseInt(viewBoxAttr[2], 10);
500         width = parseInt(viewBoxAttr[3], 10);
501         height = parseInt(viewBoxAttr[4], 10);
502       }
503 
504       // values of width/height attributes overwrite those extracted from viewbox attribute
505       width = widthAttr ? parseFloat(widthAttr) : width;
506       height = heightAttr ? parseFloat(heightAttr) : height;
507 
508       var options = {
509         width: width,
510         height: height
511       };
512 
513       fabric.gradientDefs = fabric.getGradientDefs(doc);
514       fabric.cssRules = getCSSRules(doc);
515 
516       // Precedence of rules:   style > class > attribute
517 
518       fabric.parseElements(elements, function(instances) {
519         fabric.documentParsingTime = new Date() - startTime;
520         if (callback) {
521           callback(instances, options);
522         }
523       }, clone(options));
524     };
525   })();

562    /**
563     * Takes url corresponding to an SVG document, and parses it into a set of fabric objects
564     * @method loadSVGFromURL
565     * @param {String} url
566     * @param {Function} callback
567     */
568    function loadSVGFromURL(url, callback) {
569 
570      url = url.replace(/^\n\s*/, '').trim();
571 
572      svgCache.has(url, function (hasUrl) {
573        if (hasUrl) {
574          svgCache.get(url, function (value) {
575            var enlivedRecord = _enlivenCachedObject(value);
576            callback(enlivedRecord.objects, enlivedRecord.options);
577          });
578        }
579        else {
580          new fabric.util.request(url, {
581            method: 'get',
582            onComplete: onComplete
583          });
584        }
585      });
586 
587      function onComplete(r) {
588 
589        var xml = r.responseXML;
590        if (!xml.documentElement && fabric.window.ActiveXObject && r.responseText) {
591          xml = new ActiveXObject('Microsoft.XMLDOM');
592          xml.async = 'false';
593          //IE chokes on DOCTYPE
594          xml.loadXML(r.responseText.replace(/<!DOCTYPE[\s\S]*?(\[[\s\S]*\])*?>/i,''));
595        }
596        if (!xml.documentElement) return;
597 
598        fabric.parseSVGDocument(xml.documentElement, function (results, options) {
599          svgCache.set(url, {
600            objects: fabric.util.array.invoke(results, 'toObject'),
601            options: options
602          });
603          callback(results, options);
604        });
605      }
606    }

670             ']]>',
671           '</style>',
672         '</defs>'
