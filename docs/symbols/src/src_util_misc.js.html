 77    /**
 78     * Changes value from one to another within certain period of time, invoking callbacks as value is being changed.
 79     * @method animate
 80     * @memberOf fabric.util
 81     * @param {Object} [options] Animation options
 82     * @param {Function} [options.onChange] Callback; invoked on every value change
 83     * @param {Function} [options.onComplete] Callback; invoked when value change is completed
 84     * @param {Number} [options.startValue=0] Starting value
 85     * @param {Number} [options.endValue=100] Ending value
 86     * @param {Number} [options.byValue=100] Value to modify the property by
 87     * @param {Function} [options.easing] Easing function
 88     * @param {Number} [options.duration=500] Duration of change
 89     */
 90   function animate(options) {
 91 
 92     options || (options = { });
 93 
 94     var start = +new Date(),
 95       duration = options.duration || 500,
 96       finish = start + duration, time, pos,
 97       onChange = options.onChange || function() { },
 98       abort = options.abort || function() { return false; },
 99       easing = options.easing || function(t, b, c, d) {return -c * Math.cos(t/d * (Math.PI/2)) + c + b;},
100       startValue = 'startValue' in options ? options.startValue : 0,
101       endValue = 'endValue' in options ? options.endValue : 100;
102       byValue = options.byValue || endValue - startValue;
103 
104     options.onStart && options.onStart();
105 
106     (function tick() {
107       time = +new Date();
108       currentTime = time > finish ? duration : (time - start);
109       onChange(easing(currentTime, startValue, byValue, duration));
110       if (time > finish || abort()) {
111         options.onComplete && options.onComplete();
112         return;
113       }
114       requestAnimFrame(tick);
115     })();
116   }

146     if (url) {

160   function enlivenObjects(objects, callback) {
161 
162     function getKlass(type) {
163       return fabric[fabric.util.string.camelize(fabric.util.string.capitalize(type))];
164     }
165 
166     var enlivenedObjects = [ ],
167         numLoadedAsyncObjects = 0,
168         // get length of all images
169         numTotalAsyncObjects = objects.filter(function (o) {
170           return getKlass(o.type).async;
171         }).length;
172 
173     var _this = this;
174 
175     objects.forEach(function (o, index) {
176       if (!o.type) {
177         return;
178       }
179       var klass = getKlass(o.type);
180       if (klass.async) {
181         klass.fromObject(o, function (o) {
182           enlivenedObjects[index] = o;
183           if (++numLoadedAsyncObjects === numTotalAsyncObjects) {
184             if (callback) {
185               callback(enlivenedObjects);
186             }
187           }
188         });
189       }
190       else {
191         enlivenedObjects[index] = klass.fromObject(o);
192       }
193     });
194 
195     if (numTotalAsyncObjects === 0 && callback) {
196       callback(enlivenedObjects);
197     }
198   }
