  8   /**
  9    * Removes value from an array.
 10    * Presence of value (and its position in an array) is determined via `Array.prototype.indexOf`
 11    * @static
 12    * @memberOf fabric.util
 13    * @method removeFromArray
 14    * @param {Array} array
 15    * @param {Any} value
 16    * @return {Array} original array
 17    */
 18   function removeFromArray(array, value) {
 19     var idx = array.indexOf(value);
 20     if (idx !== -1) {
 21       array.splice(idx, 1);
 22     }
 23     return array;
 24   };

 26   /**
 27    * Returns random number between 2 specified ones.
 28    * @static
 29    * @method getRandomInt
 30    * @memberOf fabric.util
 31    * @param {Number} min lower limit
 32    * @param {Number} max upper limit
 33    * @return {Number} random value (between min and max)
 34    */
 35   function getRandomInt(min, max) {
 36     return Math.floor(Math.random() * (max - min + 1)) + min;
 37   }

 77    /**
 78     * Changes value from one to another within certain period of time, invoking callbacks as value is being changed.
 79     * @method animate
 80     * @memberOf fabric.util
 81     * @param {Object} [options] Animation options
 82     * @param {Function} [options.onChange] Callback; invoked on every value change
 83     * @param {Function} [options.onComplete] Callback; invoked when value change is completed
 84     * @param {Number} [options.startValue=0] Starting value
 85     * @param {Number} [options.endValue=100] Ending value
 86     * @param {Number} [options.byValue=100] Value to modify the property by
 87     * @param {Function} [options.easing] Easing function
 88     * @param {Number} [options.duration=500] Duration of change
 89     */
 90   function animate(options) {
 91 
 92     options || (options = { });
 93 
 94     var start = +new Date(), 
 95       duration = options.duration || 500,
 96       finish = start + duration, time, pos,
 97       onChange = options.onChange || function() { },
 98       abort = options.abort || function() { return false; },
 99       easing = options.easing || function(t, b, c, d) {return -c * Math.cos(t/d * (Math.PI/2)) + c + b;},
100       startValue = 'startValue' in options ? options.startValue : 0,
101       endValue = 'endValue' in options ? options.endValue : 100;
102       byValue = options.byValue || endValue - startValue;
103 
104     options.onStart && options.onStart();
105     
106     (function tick() {
107       time = +new Date();
108       currentTime = time > finish ? duration : (time - start);
109       onChange(easing(currentTime, startValue, byValue, duration));
110       if (time > finish || abort()) {
111         options.onComplete && options.onComplete();
112         return;
113       }
114       requestAnimFrame(tick);
115     })();
116   }

118   /**
119     * requestAnimationFrame polyfill from http://paulirish.com/2011/requestanimationframe-for-smart-animating/
120     * @method requestAnimFrame
121     * @memberOf fabric.util
122     * @param {Function} callback Callback to invoke
123     * @param {DOMElement} element optional Element to associate with animation
124     */
125   var requestAnimFrame = (function() {
126     return  fabric.window.requestAnimationFrame       || 
127             fabric.window.webkitRequestAnimationFrame || 
128             fabric.window.mozRequestAnimationFrame    || 
129             fabric.window.oRequestAnimationFrame      || 
130             fabric.window.msRequestAnimationFrame     || 
131             function(callback, element) {
132               fabric.window.setTimeout(callback, 1000 / 60);
133             };
134   })();

136   /**
137     * Loads image element from given url and passes it to a callback
138     * @method loadImage
139     * @memberOf fabric.util
140     * @param {String} url URL representing an image
141     * @param {Function} callback Callback; invoked with loaded image
142     * @param {Any} context optional Context to invoke callback in
143     */
144   function loadImage(url, callback, context) {
145     if (url) {
146       var img = new Image();
147       /** @ignore */
148       img.onload = function () { 
149         callback && callback.call(context, img);
150         img = img.onload = null;
151       };
152       img.src = url;
153     }
154     else {
155       callback && callback.call(context, url);
156     }
157   }
