1 /**
  2  * @namespace
  3  */
  4 fabric.Observable = {
  5 
  6   /**
  7    * Observes specified event
  8    * @method observe
  9    * @depracated Since 0.8.34. Use `on` instead.
 10    * @param {String} eventName
 11    * @param {Function} handler
 12    */
 13   observe: function(eventName, handler) {
 14     if (!this.__eventListeners) {
 15       this.__eventListeners = { };
 16     }
 17     // one object with key/value pairs was passed
 18     if (arguments.length === 1) {
 19       for (var prop in eventName) {
 20         this.on(prop, eventName[prop]);
 21       }
 22     }
 23     else {
 24       if (!this.__eventListeners[eventName]) {
 25         this.__eventListeners[eventName] = [ ];
 26       }
 27       this.__eventListeners[eventName].push(handler);
 28     }
 29   },
 30 
 31   /**
 32    * Stops event observing for a particular event handler
 33    * @method stopObserving
 34    * @depracated Since 0.8.34. Use `off` instead.
 35    * @param {String} eventName
 36    * @param {Function} handler
 37    */
 38   stopObserving: function(eventName, handler) {
 39     if (!this.__eventListeners) {
 40       this.__eventListeners = { };
 41     }
 42     if (this.__eventListeners[eventName]) {
 43       fabric.util.removeFromArray(this.__eventListeners[eventName], handler);
 44     }
 45   },
 46 
 47   /**
 48    * Fires event with an optional options object
 49    * @method fire
 50    * @param {String} eventName
 51    * @param {Object} [options]
 52    */
 53   fire: function(eventName, options) {
 54     if (!this.__eventListeners) {
 55       this.__eventListeners = { }
 56     }
 57     var listenersForEvent = this.__eventListeners[eventName];
 58     if (!listenersForEvent) return;
 59     for (var i = 0, len = listenersForEvent.length; i < len; i++) {
 60       // avoiding try/catch for perf. reasons
 61       listenersForEvent[i](options || { });
 62     }
 63   }
 64 };
 65 
 66 /**
 67  * Alias for observe
 68  * @method observe
 69  * @memberOf fabric.Observable
 70  */
 71 fabric.Observable.on = fabric.Observable.observe;
 72 
 73 /**
 74  * Alias for stopObserving
 75  * @method off
 76  */
 77 fabric.Observable.off = fabric.Observable.stopObserving;
