  3   /**
  4    * Populates canvas with data from the specified dataless JSON
  5    * JSON format must conform to the one of `fabric.Canvas#toDatalessJSON`
  6    * @method loadFromDatalessJSON
  7    * @param {String} json JSON string
  8    * @param {Function} callback Callback, invoked when json is parsed
  9    *                            and corresponding objects (e.g: fabric.Image)
 10    *                            are initialized
 11    * @return {fabric.Canvas} instance
 12    * @chainable
 13    */
 14   loadFromDatalessJSON: function (json, callback) {
 15 
 16     if (!json) {
 17       return;
 18     }
 19 
 20     // serialize if it wasn't already
 21     var serialized = (typeof json === 'string')
 22       ? JSON.parse(json)
 23       : json;
 24 
 25     if (!serialized || (serialized && !serialized.objects)) return;
 26 
 27     this.clear();
 28 
 29     // TODO: test this
 30     this.backgroundColor = serialized.background;
 31     this._enlivenDatalessObjects(serialized.objects, callback);
 32   },
 33 
 34   /**
 35    * @method _enlivenDatalessObjects
 36    * @param {Array} objects
 37    * @param {Function} callback
 38    */
 39   _enlivenDatalessObjects: function (objects, callback) {
 40 
 41     /** @ignore */
 42     function onObjectLoaded(object, index) {
 43       _this.insertAt(object, index, true);
 44       object.setCoords();
 45       if (++numLoadedObjects === numTotalObjects) {
 46         callback && callback();
 47       }
 48     }
 49 
 50     var _this = this,
 51         numLoadedObjects = 0,
 52         numTotalObjects = objects.length;
 53 
 54     if (numTotalObjects === 0 && callback) {
 55       callback();
 56     }
 57 
 58     try {
 59       objects.forEach(function (obj, index) {
 60 
 61         var pathProp = obj.paths ? 'paths' : 'path';
 62         var path = obj[pathProp];
 63 
 64         delete obj[pathProp];
 65 
 66         if (typeof path !== 'string') {
 67           switch (obj.type) {
 68             case 'image':
 69               fabric[fabric.util.string.capitalize(obj.type)].fromObject(obj, function (o) {
 70                 onObjectLoaded(o, index);
 71               });
 72               break;
 73             default:
 74               var klass = fabric[fabric.util.string.camelize(fabric.util.string.capitalize(obj.type))];
 75               if (klass && klass.fromObject) {
 76                 // restore path
 77                 if (path) {
 78                   obj[pathProp] = path;
 79                 }
 80                 onObjectLoaded(klass.fromObject(obj), index);
 81               }
 82               break;
 83           }
 84         }
 85         else {
 86           if (obj.type === 'image') {
 87             fabric.util.loadImage(path, function (image) {
 88               var oImg = new fabric.Image(image);
 89 
 90               oImg.setSourcePath(path);
 91 
 92               fabric.util.object.extend(oImg, obj);
 93               oImg.setAngle(obj.angle);
 94 
 95               onObjectLoaded(oImg, index);
 96             });
 97           }
 98           else if (obj.type === 'text') {
 99 
100             obj.path = path;
101             var object = fabric.Text.fromObject(obj);
102             var onscriptload = function () {
103               // TODO (kangax): find out why Opera refuses to work without this timeout
104               if (Object.prototype.toString.call(fabric.window.opera) === '[object Opera]') {
105                 setTimeout(function () {
106                   onObjectLoaded(object, index);
107                 }, 500);
108               }
109               else {
110                 onObjectLoaded(object, index);
111               }
112             }
113 
114             fabric.util.getScript(path, onscriptload);
115           }
116           else {
117             fabric.loadSVGFromURL(path, function (elements, options) {
118               var object = fabric.util.groupSVGElements(elements, obj, path);
119 
120               // copy parameters from serialied json to object (left, top, scaleX, scaleY, etc.)
121               // skip this step if an object is a PathGroup, since we already passed it options object before
122               if (!(object instanceof fabric.PathGroup)) {
123                 fabric.util.object.extend(object, obj);
124                 if (typeof obj.angle !== 'undefined') {
125                   object.setAngle(obj.angle);
126                 }
127               }
128 
129               onObjectLoaded(object, index);
130             });
131           }
132         }
133       }, this);
134     }
135     catch(e) {
136       fabric.log(e.message);
137     }
138   },

140   /**
141    * Populates canvas with data from the specified JSON
142    * JSON format must conform to the one of `fabric.Canvas#toJSON`
143    * @method loadFromJSON
144    * @param {String} json JSON string
145    * @param {Function} callback Callback, invoked when json is parsed
146    *                            and corresponding objects (e.g: fabric.Image)
147    *                            are initialized
148    * @return {fabric.Canvas} instance
149    * @chainable
150    */
151   loadFromJSON: function (json, callback) {
152     if (!json) return;
153 
154     var serialized = JSON.parse(json);
155     if (!serialized || (serialized && !serialized.objects)) return;
156 
157     this.clear();
158     var _this = this;
159     this._enlivenObjects(serialized.objects, function () {
160       _this.backgroundColor = serialized.background;
161 
162       if (serialized.backgroundImage) {
163         _this.setBackgroundImage(serialized.backgroundImage, function() {
164 
165           _this.backgroundImageOpacity = serialized.backgroundImageOpacity;
166           _this.backgroundImageStretch = serialized.backgroundImageStretch;
167 
168           _this.renderAll();
169 
170           callback && callback();
171         });
172       }
173       else {
174         callback && callback();
175       }
176     });
177 
178     return this;
179   },
180 
181   /**
182    * @method _enlivenObjects
183    * @param {Array} objects
184    * @param {Function} callback
185    */
186   _enlivenObjects: function (objects, callback) {
187     var _this = this;
188     fabric.util.enlivenObjects(objects, function(enlivenedObjects) {
189       enlivenedObjects.forEach(function(obj, index) {
190         _this.insertAt(obj, index, true);
191       });
192       callback && callback();
193     });
194   },

221   /**
222    * Clones canvas instance
223    * @method clone
224    * @param {Object} [callback] Receives cloned instance as a first argument
225    */
226   clone: function (callback) {
227     var data = JSON.stringify(this);
228     this.cloneWithoutData(function(clone) {
229       clone.loadFromJSON(data, function() {
230         callback && callback(clone);
231       });
232     });
233   },

235   /**
236    * Clones canvas instance without cloning existing data.
237    * This essentially copies canvas dimensions, clipping properties, etc.
238    * but leaves data empty (so that you can populate it with your own)
239    * @method cloneWithoutData
240    * @param {Object} [callback] Receives cloned instance as a first argument
241    */
242   cloneWithoutData: function(callback) {
243     var el = fabric.document.createElement('canvas');
244 
245     el.width = this.getWidth();
246     el.height = this.getHeight();
247 
248     var clone = new fabric.Canvas(el);
249     clone.clipTo = this.clipTo;
250     if (this.backgroundImage) {
251       clone.setBackgroundImage(this.backgroundImage.src, function() {
252         clone.renderAll();
253         callback && callback(clone);
254       });
255       clone.backgroundImageOpacity = this.backgroundImageOpacity;
256       clone.backgroundImageStretch = this.backgroundImageStretch;
257     }
258     else {
259       callback && callback(clone);
260     }
261   }
