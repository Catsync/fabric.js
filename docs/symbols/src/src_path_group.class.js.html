243   /**

247   function instantiatePaths(paths) {
248     for (var i = 0, len = paths.length; i < len; i++) {
249       if (!(paths[i] instanceof fabric.Object)) {
250         var klassName = camelize(capitalize(paths[i].type));
251         paths[i] = fabric[klassName].fromObject(paths[i]);
252       }
253     }
254     return paths;
255   }
