157     /**
158      * @private
159      * @method _render
160      * @param {CanvasRenderingContext2D} ctx Context to render on

162     _render: function(ctx) {
163       var o = Cufon.textOptions || (Cufon.textOptions = { });
164 
165       // export options to be used by cufon.js
166       o.left = this.left;
167       o.top = this.top;
168       o.context = ctx;
169       o.color = this.fill;
170 
171       var el = this._initDummyElement();
172 
173       // set "cursor" to top/left corner
174       this.transform(ctx);
175 
176       // draw text
177       Cufon.replaceElement(el, {
178         engine: 'canvas',
179         separate: 'none',
180         fontFamily: this.fontFamily,
181         fontWeight: this.fontWeight,
182         textDecoration: this.textDecoration,
183         textShadow: this.textShadow,
184         textAlign: this.textAlign,
185         fontStyle: this.fontStyle,
186         lineHeight: this.lineHeight,
187         strokeStyle: this.strokeStyle,
188         strokeWidth: this.strokeWidth,
189         backgroundColor: this.backgroundColor
190       });
191 
192       // update width, height
193       this.width = o.width;
194       this.height = o.height;
195       this._totalLineHeight = o.totalLineHeight;
196       this._fontAscent = o.fontAscent;
197       this._boundaries = o.boundaries;
198       this._shadowOffsets = o.shadowOffsets;
199       this._shadows = o.shadows || [ ];
200 
201       // need to set coords _after_ the width/height was retreived from Cufon
202       this.setCoords();

280     /**
281      * Returns svg representation of an instance
282      * @method toSVG
283      * @return {string} svg representation of an instance
284      */
285     toSVG: function() {
286 
287       var textLines = this.text.split(/\r?\n/),
288           lineTopOffset = -this._fontAscent - ((this._fontAscent / 5) * this.lineHeight),
289 
290           textLeftOffset = -(this.width/2),
291           textTopOffset = (this.height/2) - (textLines.length * this.fontSize) - this._totalLineHeight,
292 
293           textAndBg = this._getSVGTextAndBg(lineTopOffset, textLeftOffset, textLines),
294           shadowSpans = this._getSVGShadows(lineTopOffset, textLines);
295 
296       // move top offset by an ascent
297       textTopOffset += ((this._fontAscent / 5) * this.lineHeight);
298 
299       return [
300         '<g transform="', this.getSvgTransform(), '">',
301           textAndBg.textBgRects.join(''),
302           '<text ',
303             (this.fontFamily ? 'font-family="\'' + this.fontFamily + '\'" ': ''),
304             (this.fontSize ? 'font-size="' + this.fontSize + '" ': ''),
305             (this.fontStyle ? 'font-style="' + this.fontStyle + '" ': ''),
306             (this.fontWeight ? 'font-weight="' + this.fontWeight + '" ': ''),
307             (this.textDecoration ? 'text-decoration="' + this.textDecoration + '" ': ''),
308             'style="', this.getSvgStyles(), '" ',
309             /* svg starts from left/bottom corner so we normalize height */
310             'transform="translate(', toFixed(textLeftOffset, 2), ' ', toFixed(textTopOffset, 2), ')">',
311             shadowSpans.join(''),
312             textAndBg.textSpans.join(''),
313           '</text>',
314         '</g>'
315       ].join('');
316     },

318     _getSVGShadows: function(lineTopOffset, textLines) {
319       var shadowSpans = [], j, i, jlen, ilen, lineTopOffsetMultiplier = 1;
320 
321       for (j = 0, jlen = this._shadows.length; j < jlen; j++) {
322         for (i = 0, ilen = textLines.length; i < ilen; i++) {
323           if (textLines[i] !== '') {
324             var lineLeftOffset = (this._boundaries && this._boundaries[i]) ? this._boundaries[i].left : 0;
325             shadowSpans.push(
326               '<tspan x="',
327               toFixed((lineLeftOffset + lineTopOffsetMultiplier) + this._shadowOffsets[j][0], 2),
328               (i === 0 ? '" y' : '" dy'), '="',
329               toFixed(lineTopOffset + (i === 0 ? this._shadowOffsets[j][1] : 0), 2),
330               '" ',
331               this._getFillAttributes(this._shadows[j].color), '>',
332               fabric.util.string.escapeXml(textLines[i]),
333             '</tspan>');
334             lineTopOffsetMultiplier = 1;
335           } else {
336             // in some environments (e.g. IE 7 & 8) empty tspans are completely ignored, using a lineTopOffsetMultiplier
337             // prevents empty tspans
338             lineTopOffsetMultiplier++;
339           }
340         }
341       }
342       return shadowSpans;
343     },
344 
345     _getSVGTextAndBg: function(lineTopOffset, textLeftOffset, textLines) {
346       var textSpans = [ ], textBgRects = [ ], i, lineLeftOffset, len, lineTopOffsetMultiplier = 1;
347 
348       // text and background
349       for (i = 0, len = textLines.length; i < len; i++) {
350         if (textLines[i] !== '') {
351           lineLeftOffset = (this._boundaries && this._boundaries[i]) ? toFixed(this._boundaries[i].left, 2) : 0;
352           textSpans.push(
353             '<tspan x="',
354             lineLeftOffset, '" ',
355             (i === 0 ? 'y' : 'dy'), '="',
356             toFixed(lineTopOffset * lineTopOffsetMultiplier, 2) , '" ',
357             // doing this on <tspan> elements since setting opacity on containing <text> one doesn't work in Illustrator
358             this._getFillAttributes(this.fill), '>',
359             fabric.util.string.escapeXml(textLines[i]),
360             '</tspan>'
361           );
362           lineTopOffsetMultiplier = 1;
363         } else {
364           // in some environments (e.g. IE 7 & 8) empty tspans are completely ignored, using a lineTopOffsetMultiplier
365           // prevents empty tspans
366           lineTopOffsetMultiplier++;
367         }
368 
369         if (!this.backgroundColor) continue;
370 
371         textBgRects.push(
372           '<rect ',
373             this._getFillAttributes(this.backgroundColor),
374             ' x="',
375             toFixed(textLeftOffset + this._boundaries[i].left, 2),
376             '" y="',
377             /* an offset that seems to straighten things out */
378             toFixed((lineTopOffset * i) - this.height / 2, 2),
379             '" width="',
380             toFixed(this._boundaries[i].width, 2),
381             '" height="',
382             toFixed(this._boundaries[i].height, 2),
383           '"></rect>');
384       }
385       return {
386         textSpans: textSpans,
387         textBgRects: textBgRects
388       };
389     },

391     // Adobe Illustrator (at least CS5) is unable to render rgba()-based fill values
392     // we work around it by "moving" alpha channel into opacity attribute and setting fill's alpha to 1
393     _getFillAttributes: function(value) {
394       var fillColor = value ? new fabric.Color(value) : '';
395       if (!fillColor || !fillColor.getSource() || fillColor.getAlpha() === 1) {
396         return 'fill="' + value + '"';
397       }
398       return 'opacity="' + fillColor.getAlpha() + '" fill="' + fillColor.setAlpha(1).toRgb() + '"';
399     },
