 20 
 21     /**
 22      * @property
 23      * @type String
 24      */
 25     type: 'line',
 26 
 27     /**
 28      * Constructor
 29      * @method initialize
 30      * @param {Array} points Array of points
 31      * @param {Object} [options] Options object
 32      * @return {fabric.Line} thisArg
 33      */
 34     initialize: function(points, options) {
 35       if (!points) {
 36         points = [0, 0, 0, 0];
 37       }
 38 
 39       this.callSuper('initialize', options);
 40 
 41       this.set('x1', points[0]);
 42       this.set('y1', points[1]);
 43       this.set('x2', points[2]);
 44       this.set('y2', points[3]);
 45 
 46       this._setWidthHeight(options);
 47     },
 48 
 49     _setWidthHeight: function(options) {
 50       options || (options = { });
 51 
 52       this.set('width', (this.x2 - this.x1) || 1);
 53       this.set('height', (this.y2 - this.y1) || 1);
 54 
 55       this.set('left', 'left' in options ? options.left : (this.x1 + this.width / 2));
 56       this.set('top', 'top' in options ? options.top : (this.y1 + this.height / 2));
 57     },
 58 
 59     set: function(name, value) {
 60       parentSet.call(this, name, value);
 61       if (name in coordProps) {
 62         this._setWidthHeight();
 63       }
 64       return this;
 65     },
 66 
 67     /**
 68      * @private
 69      * @method _render
 70      * @param {CanvasRenderingContext2D} ctx Context to render on
 71      */
 72     _render: function(ctx) {
 73       ctx.beginPath();
 74 
 75       // move from center (of virtual box) to its left/top corner
 76       ctx.moveTo(this.width === 1 ? 0 : (-this.width / 2), this.height === 1 ? 0 : (-this.height / 2));
 77       ctx.lineTo(this.width === 1 ? 0 : (this.width / 2), this.height === 1 ? 0 : (this.height / 2));
 78 
 79       ctx.lineWidth = this.strokeWidth;
 80 
 81       // TODO: test this
 82       // make sure setting "fill" changes color of a line
 83       // (by copying fillStyle to strokeStyle, since line is stroked, not filled)
 84       var origStrokeStyle = ctx.strokeStyle;
 85       ctx.strokeStyle = ctx.fillStyle;
 86       ctx.stroke();
 87       ctx.strokeStyle = origStrokeStyle;
 88     },

 97     },

101      * @methd toObject

103      */
104     toObject: function() {

106         x1: this.get('x1'),

131   /**

136   fabric.Line.ATTRIBUTE_NAMES = 'x1 y1 x2 y2 stroke stroke-width transform'.split(' ');

138   /**
139    * Returns fabric.Line instance from an SVG element
140    * @static
141    * @method fabric.Line.fromElement

144    * @return {fabric.Line} instance of fabric.Line

146   fabric.Line.fromElement = function(element, options) {
147     var parsedAttributes = fabric.parseAttributes(element, fabric.Line.ATTRIBUTE_NAMES);
148     var points = [

150       parsedAttributes.y1 || 0,
151       parsedAttributes.x2 || 0,
152       parsedAttributes.y2 || 0
153     ];
154     return new fabric.Line(points, extend(parsedAttributes, options));
155   };

157   /**

159    * @static

161    * @param {Object} object Object to create an instance from
162    * @return {fabric.Line} instance of fabric.Line
163    */
164   fabric.Line.fromObject = function(object) {
165     var points = [object.x1, object.y1, object.x2, object.y2];
166     return new fabric.Line(points, object);
